# Створення бібліотеки і виконавчого файлу в одному проекті
CMAKE_MINIMUM_REQUIRED(VERSION 3.8)

PROJECT(lab_1_cmake C)

set(CMAKE_DISABLE_IN_SOURCE_BUILD ON)
set(CMAKE_DISABLE_SOURCE_CHANGES  ON)
# SET(CMAKE_VERBOSE_MAKEFILE ON)

set(CMAKE_C_STANDARD 99)
set(CMAKE_C_EXTENSIONS FALSE)
set(CMAKE_C_FLAGS  "${CMAKE_CXX_FLAGS} -w")


file(GLOB SOURCES *.c *.h)
SET(LIB_NAME zlib)


ADD_LIBRARY(${LIB_NAME} STATIC ${SOURCES})
ADD_LIBRARY(${LIB_NAME}-share SHARED ${SOURCES})
set_target_properties(${LIB_NAME}-share PROPERTIES OUTPUT_NAME ${LIB_NAME})
